# 별 찍기 5
```java
import java.util.Scanner;
public class Main {
    public static void main(String[] args){
        
        Scanner in = new Scanner(System.in);
        
        int num = in.nextInt();
        
        for(int i=0; i<num; i++) { 
        	for(int j=0;j<num-i-1;j++){
        		System.out.print(" ");
        	}
            for(int j=0; j<i*2+1; j++) {
            
                System.out.print("*");
            }        
            System.out.println();            
        }
    }
}
```
